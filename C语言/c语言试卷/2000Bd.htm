五、编程题（第1题4分，第2题10分，共14分）

1、

# include "math.h"        头文件1分

main( )

{ float x, y;              变量说明、输入、输出 1 分

 scanf("%f", &x);     

 if(x>10) y=exp(x);        2分

 else if(x==10) y=0;        

 else y=3*x+5;          

 printf("%f\n", y);

}

 

 

2、

# include <stdio.h>         头文件、变量说明 1 分

main()

{

    FILE *fpin;

    char ch;

    if (( fpin = fopen("input.txt","r" )) == NULL){       2 分

           printf( "Can't open file input.txt\n" );

           exit(0);

    }

    while ( !feof( fpin )){                 2 分

           ch=fgetc(fpin);                  2 分

        if(ch!=' ')  putchar(ch);          2 分

    }

    fclose( fpin);                          关闭文件 1 分

}

 

或 while ((ch=fgetc(fpin))!=EOF)       4 分(读2分，判断2分)
