int  sum_fact(int number)    定义、变量说明、返回 2分

{

    int k, sum = 0;

    if ( number < 0 ) number = - number;

    for(k = 1; k <= number; k++)            1 分

           if( number%k == 0 ) sum += k;       2 分

    return sum;

}
